def frustum_axis_alignment(
    clusters: torch.Tensor,        # (K, P, 3)
    counts: torch.Tensor,          # (K,)
    frustum_angles: torch.Tensor,  # (K,)
) -> torch.Tensor:
    """
    Frustum åæ ‡è½´å¯¹é½ç®—å­ã€‚

    è¯¥ç®—å­å°†æ¯ä¸ª Frustum ç‚¹äº‘æ ¹æ®å…¶å¯¹åº”çš„è§†é”¥æ–¹å‘è§’è¿›è¡Œæ—‹è½¬ï¼Œ
